def <<(line)
  case line
  when /^\-\-\-/
    handle #@doc
    @doc << line
  when /^\.\.\./
    handle #@doc
    stop
  else
    @doc << line
  end
end

def consume(input)
  @doc  = ''
  @done = false
  while line = input.gets
    self << line
  end
  handle unless @done   # in case `...` was left out
end

def handle
  return if @doc == ''
  entry = YAML.load(@doc)
  @reporter << entry
  @doc = ''
end
